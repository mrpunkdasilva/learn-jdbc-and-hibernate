Modelo Entidade-Relacionamento

Criação do banco de dados

CREATE DATABASE BIBLIOTECA;

Criando as tabelas

  • Livros:

CREATE TABLE LIVROS ( ID SERIAL PRIMARY KEY, ISBN CHAR(13) UNIQUE NOT NULL, TITULO CHARACTER VARYING(100) NOT NULL, AUTOR CHARACTER VARYING(100) NOT NULL, ANO_PUBLICACAO DATE NOT NULL, LIVRO_EM_ESTOQUE BOOLEAN NOT NULL DEFAULT TRUE );

  • Membros:

CREATE TABLE MEMBROS ( ID SERIAL PRIMARY KEY, NOME CHARACTER VARYING(120) NOT NULL, DATA_CADASTRO DATE NOT NULL );

  • Empréstimos:

CREATE TABLE EMPRESTIMOS ( ID SERIAL PRIMARY KEY, ID_LIVRO INTEGER, CONSTRAINT FK_LIVRO FOREIGN KEY (ID_LIVRO) REFERENCES LIVROS (ID), ID_MEMBRO INTEGER, CONSTRAINT FK_MEMBRO FOREIGN KEY (ID_MEMBRO) REFERENCES MEMBROS (ID), DATA_EMPRESTIMO DATE DEFAULT CURRENT_DATE, DATA_DEVOLUCAO_PREVISTA DATE, DEVOLVIDO BOOLEAN DEFAULT FALSE );

  • Histórico de Empréstimos:

CREATE TABLE HISTORICO_EMPRESTIMOS ( ID SERIAL PRIMARY KEY, DATA_DEVOLUCAO DATE, DATA_EMPRESTIMO DATE, ID_EMPRESTIMO INTEGER, CONSTRAINT FK_EMPRESTIMO FOREIGN KEY (ID_EMPRESTIMO) REFERENCES LIVROS (ID), ID_LIVRO INTEGER, CONSTRAINT FK_EMPRESTIMO_LIVRO FOREIGN KEY (ID_LIVRO) REFERENCES LIVROS (ID), ID_MEMBRO INTEGER, CONSTRAINT FK_EMPRESTIMO_MEMBRO FOREIGN KEY (ID_MEMBRO) REFERENCES MEMBROS (ID) );
